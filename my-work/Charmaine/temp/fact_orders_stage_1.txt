WITH source_a AS (
    SELECT * FROM {{ source('brazil_ecommerce', 'olist_orders_dataset') }}
),
source_b AS (
    SELECT * FROM {{ source('brazil_ecommerce', 'olist_order_items_dataset') }}
),
source_c AS (
    SELECT * FROM {{ source('brazil_ecommerce', 'olist_order_payments_dataset') }}
),
joined_table AS (
    SELECT
        -- IDs (strings)
        orders.order_id AS id,
        orders.customer_id,

        -- categorical string
        orders.order_status,
        
        -- Timestamps with explicit UTC conversion for BigQuery
        TIMESTAMP(CAST(orders.order_purchase_timestamp AS STRING), 'UTC') AS order_purchase_timestamp,
        TIMESTAMP(CAST(orders.order_approved_at AS STRING), 'UTC') AS order_approved_at,
        TIMESTAMP(CAST(orders.order_delivered_carrier_date AS STRING), 'UTC') AS order_delivered_carrier_date,
        TIMESTAMP(CAST(orders.order_delivered_customer_date AS STRING), 'UTC') AS order_delivered_customer_date,
        TIMESTAMP(CAST(orders.order_estimated_delivery_date AS STRING), 'UTC') AS order_estimated_delivery_date,
        
        -- Integer and string from source_b
        b.order_item_id, --integer
        b.seller_id,    --string
        TIMESTAMP(CAST(b.shipping_limit_date AS STRING), 'UTC') AS shipping_limit_date,
        
        -- Numeric values
        CAST(b.price AS FLOAT64) AS price,
        CAST(b.freight_value AS FLOAT64) AS freight_value,
        
        -- Payment details
        c.payment_sequential, --integer
        COALESCE(c.payment_type, 'not_defined') AS payment_type,
        c.payment_installments, --integer
        CAST(c.payment_value AS FLOAT64) AS payment_value
        
    FROM source_a orders
    LEFT JOIN source_b b ON orders.order_id = b.order_id
    LEFT JOIN source_c c ON orders.order_id = c.order_id
    WHERE orders.order_id IS NOT NULL 
        AND orders.customer_id IS NOT NULL 
        AND b.order_item_id IS NOT NULL 
        AND b.seller_id IS NOT NULL
    QUALIFY ROW_NUMBER() OVER (PARTITION BY id ORDER BY order_purchase_timestamp DESC) = 1
),

-- 2.1 Strip whitespace and cast to proper types
cleaned_types AS (
    SELECT
        -- Strings: trim whitespace
        TRIM(CAST(id AS STRING)) AS id,
        TRIM(CAST(customer_id AS STRING)) AS customer_id,
        TRIM(CAST(order_status AS STRING)) AS order_status,
        
        -- Timestamps (already in UTC)
        order_purchase_timestamp,
        order_approved_at,
        order_delivered_carrier_date,
        order_delivered_customer_date,
        order_estimated_delivery_date,
        
        -- Integer: ensure no decimals, handle NULLs
        CAST(order_item_id AS INT64) AS order_item_id,
        
        -- Strings
        TRIM(CAST(seller_id AS STRING)) AS seller_id,
        
        -- Timestamp (already in UTC)
        shipping_limit_date,
        
        -- Floats: handle potential NULLs and invalid values
        CAST(price AS FLOAT64) AS price,
        CAST(freight_value AS FLOAT64) AS freight_value,
        
        -- Integer
        CAST(payment_sequential AS INT64) AS payment_sequential,
        
        -- String with trim
        TRIM(CAST(payment_type AS STRING)) AS payment_type,
        
        -- Integer: ensure whole number
        CAST(payment_installments AS INT64) AS payment_installments,
        
        -- Float
        CAST(payment_value AS FLOAT64) AS payment_value
        
    FROM joined_table
),

-- 2.2 Convert to consistent case for categorical columns
standardized_case AS (
    SELECT
        id,
        customer_id,
        LOWER(order_status) AS order_status,
        order_purchase_timestamp,
        order_approved_at,
        order_delivered_carrier_date,
        order_delivered_customer_date,
        order_estimated_delivery_date,
        order_item_id,
        seller_id,
        shipping_limit_date,
        price,
        freight_value,
        payment_sequential,
        LOWER(payment_type) AS payment_type,
        payment_installments,
        payment_value
    FROM cleaned_types
),

-- 2.3 Replace common null placeholders with actual NULL and validate numeric ranges
null_placeholders_replaced AS (
    SELECT
        id,
        customer_id,
        NULLIF(order_status, 'unavailable') AS order_status,
        
        -- Timestamps (keep as is, already in UTC)
        order_purchase_timestamp,
        order_approved_at,
        order_delivered_carrier_date,
        order_delivered_customer_date,
        order_estimated_delivery_date,
        
        -- For integers, ensure no weird placeholder values (negative or zero where not applicable)
        CASE 
            WHEN order_item_id <= 0 THEN NULL 
            ELSE order_item_id 
        END AS order_item_id,
        
        seller_id,
        
        -- Timestamp (keep as is, already in UTC)
        shipping_limit_date,
        
        -- For floats, handle negative or zero where price should be positive
        CASE 
            WHEN price <= 0 THEN NULL 
            ELSE price 
        END AS price,
        
        CASE 
            WHEN freight_value < 0 THEN NULL 
            ELSE freight_value 
        END AS freight_value,
        
        -- For payment sequential, should be positive
        CASE 
            WHEN payment_sequential <= 0 THEN NULL 
            ELSE payment_sequential 
        END AS payment_sequential,
        
        NULLIF(payment_type, 'not_defined') AS payment_type,
        
        -- Payment installments should be >= 0 (0 might mean no installments)
        CASE 
            WHEN payment_installments < 0 THEN NULL 
            ELSE payment_installments 
        END AS payment_installments,
        
        -- Payment value should be positive
        CASE 
            WHEN payment_value <= 0 THEN NULL 
            ELSE payment_value 
        END AS payment_value
        
    FROM standardized_case
),

-- 2.4 Add data quality flags for timestamp logic
quality_checks AS (
    SELECT
        *,
        -- Flag invalid date logic (delivery before purchase)
        CASE 
            WHEN order_delivered_customer_date IS NOT NULL 
                 AND order_purchase_timestamp IS NOT NULL
                 AND order_delivered_customer_date < order_purchase_timestamp 
            THEN TRUE
            ELSE FALSE
        END AS has_invalid_delivery_date,
        
        -- Flag overdue deliveries
        CASE 
            WHEN order_estimated_delivery_date IS NOT NULL 
                 AND CURRENT_TIMESTAMP() > order_estimated_delivery_date
                 AND order_delivered_customer_date IS NULL
            THEN TRUE
            ELSE FALSE
        END AS is_overdue_delivery,
        
        -- Flag high value products
        CASE 
            WHEN price > 10000 THEN TRUE
            ELSE FALSE
        END AS is_high_value_product
        
    FROM null_placeholders_replaced
)

-- Final SELECT
SELECT 
    id,
    customer_id,
    order_status,
    order_purchase_timestamp,
    order_approved_at,
    order_delivered_carrier_date,
    order_delivered_customer_date,
    order_estimated_delivery_date,
    order_item_id,
    seller_id,
    shipping_limit_date,
    price,
    freight_value,
    payment_sequential,
    payment_type,
    payment_installments,
    payment_value,
    has_invalid_delivery_date,
    is_overdue_delivery,
    is_high_value_product
FROM quality_checks
ORDER BY order_purchase_timestamp DESC
